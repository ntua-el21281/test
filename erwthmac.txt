// --- LCD FUNCTIONS (Για σύνδεση στο PORT A) ---
// PA2 -> RS
// PA3 -> EN
// PA4-PA7 -> D4-D7

void LCD_Pulse(void) {
    LL_GPIO_SetOutputPin(GPIOA, LL_GPIO_PIN_3); // EN High
    LL_mDelay(1);
    LL_GPIO_ResetOutputPin(GPIOA, LL_GPIO_PIN_3); // EN Low
    LL_mDelay(1);
}

void LCD_Send4Bit(uint8_t val) {
    // Καθαρισμός των Data Pins (PA4-PA7)
    LL_GPIO_ResetOutputPin(GPIOA, LL_GPIO_PIN_4 | LL_GPIO_PIN_5 | LL_GPIO_PIN_6 | LL_GPIO_PIN_7);

    // Τοποθέτηση τιμών
    if (val & 0x01) LL_GPIO_SetOutputPin(GPIOA, LL_GPIO_PIN_4);
    if (val & 0x02) LL_GPIO_SetOutputPin(GPIOA, LL_GPIO_PIN_5);
    if (val & 0x04) LL_GPIO_SetOutputPin(GPIOA, LL_GPIO_PIN_6);
    if (val & 0x08) LL_GPIO_SetOutputPin(GPIOA, LL_GPIO_PIN_7);

    LCD_Pulse();
}

void LCD_Cmd(uint8_t cmd) {
    LL_GPIO_ResetOutputPin(GPIOA, LL_GPIO_PIN_2); // RS Low (Command)
    LCD_Send4Bit(cmd >> 4); // Στείλε τα 4 High bits
    LCD_Send4Bit(cmd);      // Στείλε τα 4 Low bits
    LL_mDelay(2);
}

void LCD_Data(uint8_t data) {
    LL_GPIO_SetOutputPin(GPIOA, LL_GPIO_PIN_2); // RS High (Data)
    LCD_Send4Bit(data >> 4);
    LCD_Send4Bit(data);
    LL_mDelay(1);
}

void LCD_Clear(void) {
    LCD_Cmd(0x01); // Εντολή καθαρισμού
    LL_mDelay(2);
}

void LCD_Home(void) {
    LCD_Cmd(0x80); // Μετακίνηση στην αρχή (Line 1, Col 1)
}

void LCD_NewLine(void) {
    LCD_Cmd(0xC0); // Μετακίνηση στη 2η γραμμή
}

void LCD_Init(void) {
    LL_mDelay(50);

    // Initial Sequence
    LL_GPIO_ResetOutputPin(GPIOA, LL_GPIO_PIN_2); // RS Low
    LCD_Send4Bit(0x03); LL_mDelay(5);
    LCD_Send4Bit(0x03); LL_mDelay(1);
    LCD_Send4Bit(0x03); LL_mDelay(1);
    LCD_Send4Bit(0x02); // Γύρνα σε 4-bit mode
    LL_mDelay(1);

    // Setup
    LCD_Cmd(0x28); // 4-bit, 2 lines, 5x8 font
    LCD_Cmd(0x0C); // Display ON, Cursor OFF
    LCD_Cmd(0x06); // Increment Cursor
    LCD_Cmd(0x01); // Clear
    LL_mDelay(5);
}

void LCD_String(char *str) {
    while(*str) LCD_Data(*str++);
}

// --- UART Functions ---

void UART4_SendString(char *str) {
    while (*str) {
        while (!LL_USART_IsActiveFlag_TXE(UART4)) {}
        LL_USART_TransmitData8(UART4, *str++);
    }
}

uint8_t UART_XOR_Checksum(const char *data)
{
    uint8_t cs = 0;
    while (*data) {
        cs ^= (uint8_t)(*data++);
    }
    return cs;
}

void UART4_SendFrame(const char *payload)
{
    uint8_t cs = UART_XOR_Checksum(payload);
    char buf[128];

    // Build the message
    // Example: $TEMP=25*3A\r\n
    snprintf(buf, sizeof(buf),
             "$%s*%02X\r\n",
             payload, cs);

    UART4_SendString(buf);
}


void UART5_SendString(char *str) {
    while (*str) {
        while (!LL_USART_IsActiveFlag_TXE(UART5)) {}
        LL_USART_TransmitData8(UART5, *str++);
    }
}

uint8_t UART_VerifyFrame(const char *frame, char *payload_out)
{
    if (frame[0] != '$')
        return 0;

    char *star = strchr(frame, '*');
    if (!star)
        return 0;

    uint8_t rx_cs = (uint8_t)strtol(star + 1, NULL, 16);

    uint16_t len = star - frame - 1;
    strncpy(payload_out, frame + 1, len);
    payload_out[len] = '\0';

    uint8_t calc_cs = UART_XOR_Checksum(payload_out);

    return (rx_cs == calc_cs);
}

uint8_t UART4_ReceiveString(char *buffer, uint8_t limit, uint32_t timeout_ms)
{
    // Clear possible errors
    if (LL_USART_IsActiveFlag_ORE(UART4)) LL_USART_ClearFlag_ORE(UART4);
    if (LL_USART_IsActiveFlag_FE(UART4))  LL_USART_ClearFlag_FE(UART4);

    uint8_t i = 0;
    memset(buffer, 0, limit);

    // Rough timeout counter (CPU dependent)
    uint32_t timeout_counter = timeout_ms * 20000;

    while (timeout_counter > 0)
    {
        if (LL_USART_IsActiveFlag_RXNE(UART4)) {

            char c = (char)LL_USART_ReceiveData8(UART4);

            // Ignore NULLs
            if (c == '\0')
                continue;

            // End of line
            if (c == '\n' || c == '\r') {
                if (i > 0) {
                    buffer[i] = '\0';
                    return i;
                }
                continue;
            }

            // Store character
            if (i < limit - 1) {
                buffer[i++] = c;
            }

            // Refresh timeout after each byte
            timeout_counter = 200000;
        }

        timeout_counter--;
    }

    buffer[i] = '\0';
    return i;  // 0 if timeout
}


uint8_t UART5_ReceiveString(char *buffer, uint8_t limit, uint32_t timeout_ms)
{
    // Καθαρισμός πιθανών σφαλμάτων πριν ξεκινήσουμε
    if (LL_USART_IsActiveFlag_ORE(UART5)) LL_USART_ClearFlag_ORE(UART5);
    if (LL_USART_IsActiveFlag_FE(UART5))  LL_USART_ClearFlag_FE(UART5);

    uint8_t i = 0;
    memset(buffer, 0, limit);

    // Χρησιμοποιούμε έναν μεγάλο μετρητή για timeout αντί για LL_mDelay
    // Για να μην "τυφλώνουμε" τον επεξεργαστή.
    // Στα 170MHz, το loop τρέχει πολύ γρήγορα, άρα θέλουμε μεγάλο νούμερο.
    // Ένα πρόχειρο timeout counter (όχι ακριβές σε ms, αλλά ασφαλές).
    uint32_t timeout_counter = timeout_ms * 20000;

    while (timeout_counter > 0)
    {
        // Συνεχής έλεγχος για δεδομένα (ΧΩΡΙΣ DELAY)
        if (LL_USART_IsActiveFlag_RXNE(UART5)) {
            char c = (char)LL_USART_ReceiveData8(UART5);

            // Αγνοούμε null χαρακτήρες
            if (c == '\0') continue;

            // Έλεγχος τερματισμού (\n ή \r)
            if (c == '\n' || c == '\r') {
                if (i > 0) { // Αν έχουμε ήδη λάβει κείμενο, επιστρέφουμε
                    buffer[i] = '\0';
                    return i;
                }
                continue; // Αν είναι στην αρχή, το αγνοούμε
            }

            // Αποθήκευση στον buffer
            if (i < limit - 1) {
                buffer[i++] = c;
            }

            // Κάθε φορά που βρίσκουμε χαρακτήρα, ανανεώνουμε λίγο το timeout
            // για να προλάβουμε τον επόμενο χαρακτήρα του μηνύματος
            timeout_counter = 200000;
        }

        timeout_counter--;
    }

    buffer[i] = '\0';
    return i;
}

{

  */

  /* USER CODE BEGIN Init */

  /* USER CODE END Init */

  /* Configure the system clock */
  SystemClock_Config();

  /* USER CODE BEGIN SysInit */

  /* USER CODE END SysInit */

  /* Initialize all configured peripherals */
  MX_GPIO_Init();
  MX_UART4_Init();
  MX_UART5_Init();
  /* USER CODE BEGIN 2 */

    // 1. Ενεργοποίηση UART (Απαραίτητο για LL)
    LL_USART_Enable(UART4);
    LL_USART_Enable(UART5);
    LL_mDelay(100);

    // 2. Έλεγχος Οθόνης LCD
    LCD_Init();
    LCD_Clear();
    LCD_String("lcd_OK");

    // 3. Μήνυμα στο PC (Debug)
    UART4_SendString("\r\n=== STM32 Start ===\r\n");

  /* USER CODE END 2 */

  /* Infinite loop */
  /* USER CODE BEGIN WHILE */
    char rx_buf[100];
    char payload[100];

    while (1)
    {
        memset(rx_buf, 0, sizeof(rx_buf));

        // Receive from PC via UART4
        if (UART4_ReceiveString(rx_buf, sizeof(rx_buf), 10000)) {

            UART4_SendString("\r\nRX: [");
            UART4_SendString(rx_buf);
            UART4_SendString("]\r\n");

            // Check checksum
            if (UART_VerifyFrame(rx_buf, payload)) {

                UART4_SendString("CHECKSUM OK | PAYLOAD: ");
                UART4_SendString(payload);
                UART4_SendString("\r\n");

                LCD_Clear();
                LCD_String("CS OK");

            } else {

                UART4_SendString("CHECKSUM ERROR\r\n");

                LCD_Clear();
                LCD_String("CS ERROR");
            }
        }

        LL_mDelay(50);
    }

  /* USER CODE END 3 */
}

/**
  * @brief System Clock Configuration
  * @retval None
  */
void SystemClock_Config(void)
{
  LL_FLASH_SetLatency(LL_FLASH_LATENCY_4);
  while(LL_FLASH_GetLatency() != LL_FLASH_LATENCY_4) {}

  LL_PWR_EnableRange1BoostMode();
  LL_RCC_HSI_Enable();
  while(LL_RCC_HSI_IsReady() != 1) {}

  LL_RCC_HSI_SetCalibTrimming(64);
  LL_RCC_PLL_ConfigDomain_SYS(LL_RCC_PLLSOURCE_HSI, LL_RCC_PLLM_DIV_4, 85, LL_RCC_PLLR_DIV_2);
  LL_RCC_PLL_EnableDomain_SYS();
  LL_RCC_PLL_Enable();
  while(LL_RCC_PLL_IsReady() != 1) {}

  LL_RCC_SetSysClkSource(LL_RCC_SYS_CLKSOURCE_PLL);
  LL_RCC_SetAHBPrescaler(LL_RCC_SYSCLK_DIV_2); // 170MHz / 2 = 85MHz System Clock
  while(LL_RCC_GetSysClkSource() != LL_RCC_SYS_CLKSOURCE_STATUS_PLL) {}

  /* Ρύθμιση περιφερειακών */
  LL_RCC_SetAHBPrescaler(LL_RCC_SYSCLK_DIV_1);
  LL_RCC_SetAPB1Prescaler(LL_RCC_APB1_DIV_1);
  LL_RCC_SetAPB2Prescaler(LL_RCC_APB2_DIV_1);

  LL_Init1msTick(170000000);
  LL_SetSystemCoreClock(170000000); // Ενημέρωση της global μεταβλητής για να ξέρει η UART
}
